/**

Karo.Buy = function() {
    var object = {
        /**
         * @property {jQuery} $el
         */
        $el: null,
        /**
         * @property {Object} options
         */
        options: {
            defaultStep: 'chooseFilm',
            defaultState: {
                sessionId: null,
                film: null,
                cinema: null
//					date: '01.11.2014'
            }
        },
        step: null,
        state: {
        },
        expanded: false,
        picker: null,
        $preloaderLayer: null,
        construct: function($el, $buttonEl, $preloader, actionUrls, options) {
            var me = this;
            me.$el = $el;
            me.actionUrls = actionUrls;
            for (var k in options) {
                me.options[k] = options[k];
            }

            me.$preloaderLayer = $('#wiPreloaderLayer');

            me.reset();

            me.$el = $('<div></div>');
            me.adoptScale();
            $('body').append(me.$el);

            $('body').on('click', '.karo-buy-button', function(e) {
                e.preventDefault();

                var step = $(this).attr('data-step');

                me.reset();
                if (step === 'default') {
                    me.currentStep();
                } else if(step === 'chooseSession') {
                    if ($(this).is('[data-film-id]')) {
                        me.state.film = $(this).attr('data-film-id');
                    }
                    if ($(this).is('[data-date]')) {
                        me.state.date = $(this).attr('data-date');
                    }
                    me.chooseSession(me.state);
                }
            });
            $(window).on('resize',function(){
                if(me.$el.children().length) {
                    me.adoptScale();
                }
            });
            /*$buttonEl.unbind('click').bind('click', function(event) {
                event.preventDefault();
                me.reset();
                me.expand();
            });*/

        },
        reset: function() {
            var me = this;
            me.step = me.options.defaultStep;
            me.state = {};
            for (var i in me.options.defaultState) {
                me.state[i] = me.options.defaultState[i];
            }
        },
        expand: function() {
            var me = this;
            me.expanded = true;
            me._initState();
        },
        fold: function() {
            var me = this;
            $('html').removeClass('disabled');
            me.$el.empty();
            me.$el.hide();
        },
        _initState: function() {
            var me = this;
            if (me.step === 'chooseFilm') {
                me.chooseFilm(me.state);
            } else if (me.step === 'chooseSession') {
                me.chooseSession(me.state);
            } else if (me.step === 'choosePlace') {
                me.choosePlace(me.state);
            } else if (me.step === 'payment') {
                me.payment(me.state);
            }
        },
        prevStep: function() {
            var me = this;
            if (me.step === 'payment') {
                me.choosePlace(me.state);
            } else if (me.step === 'choosePlace') {
                me.chooseSession(me.state);
            } else if (me.step === 'chooseSession') {
                me.chooseFilm(me.state);
            }
        },
        nextStep: function() {
            var me = this;
            if (me.step === 'chooseFilm') {
                me.chooseSession(me.state);
            } else if (me.step === 'chooseSession') {
                me.choosePlace(me.state);
            } else if (me.step === 'choosePlace') {
                me.payment(me.state);
            }
        },
        _preloaderActive: function(isActive) {
            var me = this;
            if (isActive) {
                me.$el.empty();
                me.$el.append(me.$preloaderLayer.show());
            } else {
                me.$preloaderLayer.detach();
            }

                'min-height':($(window).height()-fixed_top.height()-fixed_bottom.height())+'px'

                    cursorborderradius:0,
                    cursorborder:'0px solid #303030',
                    cursorcolor:'#303030',
                    autohidemode:false
                });
            }

            //some actions to adopt
            //$.fn.rhash('get','ticketing');
        },
        adoptScale: function(){
            var me=this,
                $body=$('body'),
                width=$body.width();
            if((width>500)&&(width<768))
                var cnt=1;
            else {
                if(width<500)
                    var cnt=0.3;
                else
                    var cnt=0.1;
            }
            var scaleSource = $body.width(),
                scaleFactor = cnt,
                maxScale = 100,
                minScale = 30;

            var fontSize = scaleSource * scaleFactor;

            if (fontSize > maxScale) fontSize = maxScale;
            if (fontSize < minScale) fontSize = minScale;

            me.$el.css({
                'font-size' :   fontSize + '%',
                'width'     :   '100%',
                'height'    :   '100%',
                'overflow'  :   'hidden',
                'position'  :   'absolute'
            }).addClass('outerPopup');
        },
        close: function(){
            var me = this;
            $('html').removeClass('disabled');
            me.$el.empty();
            me.$el.hide();
            me.reset();
            me.expanded = false;
            me.$el.removeClass('active');
            if (typeof hash === 'object' && hash.length) {
                hash.clear();
            }
        },
        _load: function(url) {
            var me = this;

            $('html').addClass('disabled');
            me._preloaderActive(true);

            me.$el.show();
            me.$el.addClass('active');
            $.ajax({
                url: url,
                success: function(data) {
                    me._preloaderActive(false);

                            hp.removeClass('ovx').addClass('oh');
